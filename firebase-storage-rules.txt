// Règles Firebase Storage à copier dans la console Firebase
// Va dans Firebase Console > Storage > Rules et colle ces règles :

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Règles pour les fichiers de devis des entreprises
    match /entreprises/{entrepriseId}/devis/{devisId}/{fileName} {
      // Permettre la lecture et l'écriture pour tous (à adapter selon tes besoins de sécurité)
      allow read, write: if true;
      
      // Ou avec authentification (si tu ajoutes l'auth plus tard) :
      // allow read, write: if request.auth != null;
      
      // Validation des types de fichiers
      allow write: if resource == null && 
        request.resource.contentType.matches('application/pdf|application/.*word.*|image/.*');
      
      // Limite de taille (10MB)
      allow write: if resource == null && 
        request.resource.size < 10 * 1024 * 1024;
    }
    
    // Bloquer tout autre accès
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
